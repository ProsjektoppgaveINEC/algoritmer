\begin{algorithm}[H]
  \KwIn{Et array $\A$ som representerer en heap med $n$ elementer, og et element $x$}
  \KwOut{Et array som representerer en heap, som inneholder $x$}
  \proc{$\Insert(\A, x)$}{
    $\A{n} \gets x$ \;
    $i \gets n$ \;
    \While{$0 < i$ \And $\A{i} < \A{$\lfloor (i-1)/2 \rfloor$}$}{
      $\A{i}, \A{$\lfloor (i-1)/2 \rfloor$} \gets \A{$\lfloor (i-1)/2 \rfloor$}, \A{i}$ \;
      $i \gets \lfloor (i-1)/2 \rfloor$ \;
    }
  }
  \caption{Innsetting i heap}
\end{algorithm}
