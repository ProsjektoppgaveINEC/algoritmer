\begin{algorithm}[H]
  \KwIn{En node $v$ og et element $x$}
  \KwOut{Dersom $x$ forekommer i en node $u$ som en etterkommer av $v$,
    fjern $u$.}
  \proc{$\Remove(v, x)$}{
    \uIf{$v = \Null$}{
      \Return \Null
    }
    \uIf{$x < v.\Element$}{
      $v.\Left \gets \Remove(v.\Left, x)$ \;
    }
    \uElseIf{$x > v.\Element$}{
      $v.\Right \gets \Remove(v.\Right, x)$ \;
    }
    \uElseIf{$v.\Left = \Null$}{
      $v \gets v.\Right$ \;
    }
    \uElseIf{$v.\Right = \Null$}{
      $v \gets v.\Left$ \;
    }
    \Else{
      $u \gets \FindMin(v.\Right)$ \;
      $v.\Element \gets u.\Element$ \;
      $v.\Right \gets \Remove(v.\Right, u.\Element)$ \;
    }
    $v.\Height \gets 1 + max(\HEIGHT(v.\Left), \HEIGHT(v.\Right))$ \;
    \Return $\Balance(v)$ \;
  }
  \caption{Sletting i et AVL-tre}
\end{algorithm}
