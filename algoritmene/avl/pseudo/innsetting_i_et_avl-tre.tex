\begin{algorithm}[H]
  \KwIn{En node $v$ og et element $x$}
  \KwOut{En oppdatert node $v$ der en node som inneholder $x$ er en
    etterkommer av $v$}
  \proc{$\Insert(v, x)$}{
    \uIf{$v = \Null$}{
      $v \gets \New\ \Node(x)$
    }
    \uElseIf{$x < v.\Element$}{
      $v.\Left \gets \Insert(v.\Left, x)$
    }
    \uElseIf{$x > v.\Element$}{
      $v.\Right \gets \Insert(v.\Right, x)$
    }
    $v.\Height \gets 1 + max(\HEIGHT(v.\Left), \HEIGHT(v.\Right))$ \;
    \Return $\Balance(v)$
  }
  \caption{Innsetting i et AVL-tre}
\end{algorithm}
