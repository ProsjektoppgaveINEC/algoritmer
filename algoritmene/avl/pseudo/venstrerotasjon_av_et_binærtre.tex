\begin{algorithm}[H]
  \KwIn{En node $z$}
  \KwOut{Roter treet til venstre, slik at $y$ blir den nye roten}
  \proc{$\LeftRotate(z)$}{
    $y \gets z.\Right$ \;
    $T_1 \gets y.\Left$ \;
    \;
    $y.\Left \gets z$ \;
    $z.\Right \gets T_1$ \;
    \;
    $z.\Height \gets 1 + max(\HEIGHT(z.\Left), \HEIGHT(z.\Right))$ \;
    $y.\Height \gets 1 + max(\HEIGHT(y.\Left), \HEIGHT(y.\Right))$ \;
    \;
    \Return $y$
  }

\end{algorithm}
