\begin{algorithm}[H]
  \KwIn{Et array $\A$ med $n$ elementer}
  \KwOut{Et \emph{sortert} array med de samme $n$ elementene}
  \proc{$\MergeSort(\A)$}{
    \uIf{$n \leq 1$}{
      \Return $\A$
    }
    $i \gets \lfloor n/2 \rfloor$\;
    $\Aone \gets \MergeSort{\A{0..i-1}}$ \;
    $\Atwo \gets \MergeSort{\A{i..n-1}}$ \;
    \Return \Merge{\Aone, \Atwo, \A}
  }
  \caption{Merge sort}
\end{algorithm}
