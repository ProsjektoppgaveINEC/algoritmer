\begin{algorithm}[H]
  \KwIn{Et array $\A$ med $n$ elementer}
  \KwOut{Et \emph{sortert} array med de samme $n$ elementene}
  \proc{$\SelectionSort(\A)$}{
    \For{$i \gets 0$ \KwTo $n - 1$}{
      $k \gets i$ \;
      \For{$j \gets i+1$ \KwTo $n - 1$}{
        \uIf{$\A{j} < \A{k}$}{
          $k \gets j$
        }
      }
      \uIf{$i \not= k$}{
        $\A{i}, \A{k} \gets \A{k}, \A{i}$
      }
    }
  }
  \caption{Selection sort}
\end{algorithm}
