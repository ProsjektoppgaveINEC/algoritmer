\begin{algorithm}[H]
  \KwIn{En (uferdig) heap $\A$ med $n$ elementer der $i$ er roten}
  \KwOut{En mindre uferdig heap}
  \proc{$\BubbleDown(\A, i, n)$}{
    $\var{largest} \gets i$\;
    $\var{left} \gets 2i + 1$\;
    $\var{right} \gets 2i + 2$\;
    \;
    \uIf{$\var{left} < n$ \And $\A{\var{largest}} < \A{\var{left}}$}{
      $\var{largest}, \var{left} \gets \var{left}, \var{largest}$
    }
    \;
    \uIf{$\var{right} < n$ \And $\A{\var{largest}} < \A{\var{right}}$}{
      $\var{largest}, \var{right} \gets \var{right}, \var{largest}$
    }
    \;
    \uIf{$i \not= \var{largest}$}{
      $\A{i}, \A{\var{largest}} \gets \A{\var{largest}}, \A{i}$\;
      $\BubbleDown{\A, \var{largest}, n}$
    }
  }

\end{algorithm}
