\begin{algorithm}[H]
  \KwIn{Et array $\A$ med $n$ elementer, $\var{low}$ og $\var{high}$ er indekser}
  \KwOut{Flytter elementer som er hhv. mindre og større til venstre og
    høyre enn en gitt index som returneres}
  \proc{$\Partition(\A, \var{low}, \var{high})$}{
    $p \gets \ChoosePivot(\A, \var{low}, \var{high})$ \;
    $\A{p}, \A{\var{high}} \gets \A{\var{high}}, \A{p}$ \;
    \;
    $\var{pivot} \gets \A{\var{high}}$ \;
    $\var{left} \gets \var{low}$ \;
    $\var{right} \gets \var{high} - 1$ \;
    \;
    \While{$\var{left} \leq \var{right}$}{
      \While{$\var{left} \leq \var{right}$ \And $\A{\var{left}} \leq \var{pivot}$}{
        $\var{left} \gets \var{left} + 1$ \;
      }
      \;
      \While{$\var{right} \geq \var{left}$ \And $\A{\var{right}} \geq \var{pivot}$}{
        $\var{right} \gets \var{right} - 1$ \;
      }
      \;
      \uIf{$\var{left} < \var{right}$}{
        $\A{\var{left}}, \A{\var{right}} \gets \A{\var{right}}, \A{\var{left}}$ \;
      }
    }
    \;
    $\A{\var{left}}, \A{\var{high}} \gets \A{\var{high}}, \A{\var{left}}$ \;
    \Return $\var{left}$
  }
  \caption{Partition}
\end{algorithm}
