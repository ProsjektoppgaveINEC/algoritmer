\begin{algorithm}[H]
  \KwIn{Et array $\A$ med $n$ elementer}
  \KwOut{Et array med de samme $n$ elementene \emph{sortert etter nøkler}}
  \proc{$\BucketSort(\A)$}{
    La $\B$ være et array med $N$ tomme lister \;
    \For{$i \gets 0$ \KwTo $n - 1$}{
      La $k$ være nøkkelen assosiert med $\A{i}$ \;
      Legg til $\A{i}$ på slutten av listen $\B{k}$
    }
    $j \gets 0$ \;
    \For{$k \gets 0$ \KwTo $N - 1$}{
      \For{hver $x$ i listen $\B{k}$}{
        $\A{j} \gets x$ \;
        $j \gets j + 1$ \;
      }
    }
    \Return \A
  }
  \caption{Bucket sort}
\end{algorithm}
