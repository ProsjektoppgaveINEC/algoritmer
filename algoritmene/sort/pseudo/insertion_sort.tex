\begin{algorithm}[H]
  \KwIn{Et array $\A$ med $n$ elementer}
  \KwOut{Et \emph{sortert} array med de samme $n$ elementene}
  \proc{$\InsertionSort(\A)$}{
    \For{$i \gets 1$ \KwTo $n - 1$}{
      $j \gets i$ \;
      \While{$j > 0$ \And $\A{j-1} > \A{j}$}{
        $\A{j-1}, \A{j} \gets \A{j}, \A{j-1}$ \;
        $j \gets j - 1$
      }
    }
  }
  \caption{Insertion sort}
\end{algorithm}
