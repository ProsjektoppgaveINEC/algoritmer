\begin{algorithm}[H]
  \KwIn{En graf $G = (V, E)$}
  \KwOut{Returner den reverserte grafen $G^{-1} = (V, E^{-1})$}
  \proc{\ReverseGraph{$G$}}{
    $E^{-1} \gets$ empty set\;
    \For{$(u, v) \in E$}{
      add $(v, u)$ to $E^{-1}$
    }
    \Return $(V, E^{-1})$
  }
  \caption{Reverser en gitt graf}
\end{algorithm}
