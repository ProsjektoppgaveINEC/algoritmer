\begin{algorithm}[H]
  \KwIn{En graf sammenhengende $G = (V, E)$}
  \KwOut{Returnerer alle seperasjonsnoder i $G$}
  $\var{depth} \gets $ empty map\;
  $\var{low} \gets $ empty map\;
  $\var{seps} \gets $ empty set\;
  \proc{\SeparationVertices{$G$}}{
    $s \gets$ choose arbitrary vertex from $V$\;
    $\var{depth}[s] \gets 0$\;
    $\var{low}[s] \gets 0$\;
    \;
    \For{$(s, u) \in E$}{
      \If{$u \not\in \var{depth}$}{
        \SeparationVerticesRec{$G$, $u$, $1$}
      }
    }
    \;
    \If{more than one $u$, such that $(s, u) \in E$ and $\var{depth}[u] = 1$}{
      add $s$ to \var{seps}\;
    }
    \Return \var{seps}
  }
  \;
  \proc{\SeparationVerticesRec{$G$, $u$, $d$}}{
    $\var{depth}[u] \gets d$ \;
    $\var{low}[u] \gets d$\;

    \For{$(u, v) \in E$}{
      \If{$v \in \var{depth}$}{
        $\var{low}[u] \gets \min(\var{low}[u], \var{depth}[u])$\;
        \Continue
      }
      \SeparationVerticesRec{$G$, $v$, $d + 1$}\;
      $\var{low}[u] \gets \min(\var{low}[u], \var{low}[v])$\;
      \If{$d \leq \var{low}[v]$}{
        add $u$ to \var{seps}\;
      }
    }
  }
  \caption{Finn alle seperasjonsnoder i en sammenhengende graf}
\end{algorithm}
