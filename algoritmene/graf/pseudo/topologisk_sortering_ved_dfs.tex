\begin{algorithm}[H]
  \KwIn{En rettet asyklisk graf $G = (V, E)$}
  \KwOut{En topologisk ordning av nodene i $G$}
  \proc{\DFSTopSort{$G$}}{
    $\var{stack} \gets $ empty stack\;
    $\var{visited} \gets $ empty set\;
    \For{$u \in V$}{
      \If{$u \not\in \var{visited}$}{
        \DFSVisit{$G$, $u$, \var{visited}, \var{stack}}
      }
    }
    \Return \var{stack}
  }
  \;
  \proc{\DFSVisit{$G$, $u$, \var{visited}, \var{stack}}}{
    add $u$ to \var{visited}\;
    \For{$(u, v) \in E$}{
      \If{$v \not\in \var{visited}$}{
        \DFSVisit{$G$, $v$, \var{visited}, \var{stack}}
      }
    }
    \var{stack}.\Push{$u$}
  }
  \caption{Topologisk sortering ved DFS}
\end{algorithm}
