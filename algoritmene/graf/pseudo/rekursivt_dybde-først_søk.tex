\begin{algorithm}[H]
  \KwIn{En graf $G = (V, E)$ og en startnode $u$}
  \KwOut{Besøker alle noder i $G$ som kan nås fra $u$ nøyaktig én gang}
  \proc{\DFSVisit{$G$, $u$, \var{visited}}}{
    add $u$ to \var{visited}\;
    \For{$(u, v) \in E$}{
      \uIf{$v \not\in \var{visited}$}{
        \DFSVisit{$G$, $v$, \var{visited}}
      }
    }
  }
  \caption{Rekursivt dybde-først søk}
\end{algorithm}
