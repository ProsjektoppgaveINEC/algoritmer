\begin{algorithm}[H]
  \KwIn{En graf $G = (V, E)$}
  \KwOut{Besøker alle noder i $G$ nøyaktig én gang bredde først}
  \proc{\BFSF{$G$}}{
    $\var{visited} \gets$ empty set\;
    \For{$v \in V$}{
      \uIf{$v \not\in \var{visited}$}{
        \BFSV{$G$, $v$, \var{visited}}
      }
    }
  }
  \caption{Fullt bredde-først søk}
\end{algorithm}
