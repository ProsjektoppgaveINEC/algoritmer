\begin{algorithm}[H]
  \KwIn{En graf $G = (V, E)$}
  \KwOut{Besøker alle noder i $G$ nøyaktig én gang dybde først}
  \proc{\DFSFull{$G$}}{
    $\var{visited} \gets$ empty set\;
    \For{$v \in V$}{
      \uIf{$v \not\in \var{visited}$}{
        \DFSVisit{$G$, $v$, \var{visited}}
      }
    }
  }
  \caption{Fullt rekursivt dybde-først søk}
\end{algorithm}
