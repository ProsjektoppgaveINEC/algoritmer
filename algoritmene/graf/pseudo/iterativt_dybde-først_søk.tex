\begin{algorithm}[H]
  \KwIn{En graf $G = (V, E)$ og en startnode $s$}
  \KwOut{Besøker alle noder i $G$ som kan nås fra $s$ nøyaktig én gang}
  \proc{\DFSVisit{$G$, $s$, \var{visited}}}{
    $\var{stack} \gets$ singleton stack containing $s$\;
    \While{\var{stack} is not empty}{
      $u \gets \var{stack}.\Pop{}$\;
      \uIf{$u \not\in \var{visited}$}{
        add $u$ to \var{visited}\;
        \For{$(u, v) \in E$}{
          \var{stack}.\Push{$v$} \;
        }
      }
    }
  }
  \caption{Iterativt dybde-først søk}
\end{algorithm}
