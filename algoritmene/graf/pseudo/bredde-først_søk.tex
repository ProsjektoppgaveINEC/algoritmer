\begin{algorithm}[H]
  \KwIn{En graf $G = (V, E)$ og en startnode $s$}
  \KwOut{Besøker alle noder i $G$ som kan nås fra $s$ nøyaktig én gang}
  \proc{\BFSV{$G$, $s$, \var{visited}}}{
    $\var{queue} \gets$ singleton queue containing $s$\;
    \While{\var{queue} is not empty}{
      $u \gets \var{queue}.\Dequeue{}$\;
      \uIf{$u \not\in \var{visited}$}{
        add $u$ to \var{visited}\;
        \For{$(u, v) \in E$}{
          \var{queue}.\Enqueue{$v$}\;
        }
      }
    }
  }
  \caption{Bredde-først søk}
\end{algorithm}
