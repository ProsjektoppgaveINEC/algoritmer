\begin{algorithm}[H]
  \KwIn{En sammenhengende graf $G = (V, E)$}
  \KwOut{Gir \KwTrue hvis grafen er bikonnektiv, \KwFalse ellers}
  \proc{\IsBiconnectedNaive{$G$}}{
    \For{$v \in V$}{
      $G' = (V', E') \gets G$ with $v$ removed\;
      $\var{visited} \gets $ empty set \;
      $u \gets $ any vertex $u \in V$ where $u \not= v$\;
      \DFSVisit{$G', u, \var{visited}$}\;
      \If{$\var{visited} \not= V'$}{
        \Return \KwFalse
      }
    }
    \Return \KwTrue\;
  }

\end{algorithm}
