\begin{algorithm}[H]
  \KwIn{En node $v$ og et element $x$}
  \KwOut{Dersom $x$ forekommer i en node $u$ som en etterkommer av $v$,
    fjern $u$.}
  \proc{$\Remove(v, x)$}{
    \uIf{$v = \Null$}{
      \Return \Null
    }
    \uIf{$x < v.\Element$}{
      $v.\Left \gets \Remove(v.\Left, x)$ \;
      \Return $v$ \;
    }
    \uIf{$x > v.\Element$}{
      $v.\Right \gets \Remove(v.\Right, x)$ \;
      \Return $v$ \;
    }
    \uIf{$v.\Left = \Null$}{
      % $u \gets v.\Right$ \;
      % $v \gets \Null$ \;
      % \Return $u$ \;
      \Return $v.\Right$ \;
    }
    \uIf{$v.\Right = \Null$}{
      % $u \gets v.\Left$ \;
      % $v \gets \Null$ \;
      % \Return $u$ \;
      \Return $v.\Left$ \;
    }
    $u \gets \FindMin(v.\Right)$ \;
    $v.\Element \gets u.\Element$ \;
    $v.\Right \gets \Remove(v.\Right, u.\Element)$ \;
    \Return $v$ \;
  }
  \caption{Slett en node i et binært søketre}
\end{algorithm}
