\begin{algorithm}[H]
  \KwIn{En node $v$ og et element $x$}
  \KwOut{Dersom $x$ forekommer i en node $u$ som en etterkommer av $v$, returner $u$, ellers \Null.}
  \proc{$\Search(v, x)$}{
    \uIf{$v = \Null$}{
      \Return \Null
    }
    \uIf{$v.\Element = x$}{
      \Return $v$
    }
    \uIf{$x < v.\Element$}{
      \Return $\Search(v.\Left, x)$
    }
    \uIf{$x > v.\Element$}{
      \Return $\Search(v.\Right, x)$
    }
  }
  \caption{Oppslag i et binært søketre}
\end{algorithm}
