\begin{algorithm}[H]
  \KwIn{En node $v$}
  \KwOut{Høyden av noden}
  \proc{$\Height(v)$}{
    $h \gets -1$ \;
    \uIf{$v = \Null$}{
      \Return $h$
    }
    \For{$v' \in v.\Children$}{
      $h \gets \Max(h, \Height(v'))$ \;
    }
    \Return $1 + h$
  }
  \caption{Finn høyden av en gitt node}
\end{algorithm}
